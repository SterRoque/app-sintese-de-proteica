    public void gerarFita(View view){
        char[] dnaFita = {'A', 'T', 'C', 'G'};
        char[] fitaGerada = new char[23];

        String[] codonFinal = {"ATT", "ATC", "ACT"};

        for (int i = 0; i < fitaGerada.length; i++){
            int indice = new Random().nextInt(4);
            int indice2 = new Random().nextInt(5);
            fitaGerada[indice2] = 'T';
            fitaGerada[indice2 + 1] = 'A';
            fitaGerada[indice2 + 2] = 'C';

            fitaGerada[i] = dnaFita[indice];
        }
        fita = String.valueOf(fitaGerada);
        dna.setText(fita);
